  <script>

    function updateLabel(id) {

      const value = document.getElementById(id).value;

      const label = document.getElementById(id + '-label');

      if (value == 1) {

        label.textContent = 'Low';

        label.className = 'indicator low';

      } else if (value == 2) {

        label.textContent = 'Medium';

        label.className = 'indicator medium';

      } else {

        label.textContent = 'High';

        label.className = 'indicator high';

      }

    }


    function setPreset(taak, context, zelfstandigheid) {

      document.getElementById('taak').value = taak;

      document.getElementById('context').value = context;

      document.getElementById('zelfstandigheid').value = zelfstandigheid;

      updateLabel('taak');

      updateLabel('context');

      updateLabel('zelfstandigheid');

    }


    window.onload = function () {

      updateLabel('taak');

      updateLabel('context');

      updateLabel('zelfstandigheid');

    }

  </script>
